```sh
# npm
npm install --save xy-messagebox

# yarn
yarn add xy-messagebox
```

## 使用

```ts
import React from "react";
import ReactDOM from "react-dom";
import XyMessagebox from "xy-messagebox";
ReactDOM.render(<XyMessagebox />, container);
```

## 开发

```sh
yarn run start
```
